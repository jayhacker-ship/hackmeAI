// Secure Banking Microservice
@RestController
@RequestMapping("/api/transactions")
public class TransactionController {
    
    @Autowired
    private TransactionService transactionService;
    
    @PostMapping("/transfer")
    @Secured("ROLE_USER")
    public ResponseEntity transferFunds(
            @Valid @RequestBody TransferRequest request) {
        
        TransactionResult result = transactionService
            .processTransfer(request);
            
        return ResponseEntity.ok(result);
    }
    
    @GetMapping("/history/{accountId}")
    @Secured("ROLE_USER")
    public Page getHistory(
            @PathVariable String accountId,
            @RequestParam(defaultValue = "0") int page) {
        return transactionService.getHistory(accountId, page);
    }
}

-- Advanced Analytics Query for E-commerce
WITH customer_segments AS (
  SELECT 
    customer_id,
    CASE 
      WHEN total_spent > 10000 THEN 'Premium'
      WHEN total_spent > 5000 THEN 'Gold'
      WHEN total_spent > 1000 THEN 'Silver'
      ELSE 'Bronze'
    END AS segment,
    total_spent,
    COUNT(DISTINCT order_id) AS order_count,
    MAX(order_date) AS last_order_date
  FROM (
    SELECT 
      c.customer_id,
      o.order_id,
      o.order_date,
      SUM(oi.price * oi.quantity) AS total_spent
    FROM customers c
    JOIN orders o ON c.customer_id = o.customer_id
    JOIN order_items oi ON o.order_id = oi.order_id
    WHERE o.order_date >= CURRENT_DATE - INTERVAL '1 year'
    GROUP BY c.customer_id, o.order_id, o.order_date
  ) AS customer_orders
  GROUP BY customer_id, total_spent
)
SELECT 
  cs.customer_id,
  c.first_name,
  c.last_name,
  c.email,
  cs.segment,
  cs.total_spent,
  cs.order_count,
  cs.last_order_date
FROM customer_segments cs
JOIN customers c ON cs.customer_id = c.customer_id
WHERE cs.segment IN ('Premium', 'Gold')
ORDER BY cs.total_spent DESC;
